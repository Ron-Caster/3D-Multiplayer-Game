// Game state let socket; let localPlayerId; let players = {}; let canvas; let ctx; let mouseX = 0; let mouseY = 0; let keys = {}; let health = 100; // Player movement const PLAYER_SPEED = 5; const PLAYER_SIZE = 30; // Start game when nickname is submitted document.getElementById('nickname-form').addEventListener('submit', (e) => { e.preventDefault(); const nicknameInput = document.getElementById('nickname-input'); const nickname = nicknameInput.value.trim(); const errorMessage = document.getElementById('error-message'); if (nickname.length < 3 || nickname.length > 15) { errorMessage.textContent = 'Nickname must be between 3 and 15 characters.'; return; } startGame(nickname); }); function startGame(nickname) { // Hide nickname form and show game document.getElementById('nickname-container').style.display = 'none'; document.getElementById('game-container').style.display = 'block'; // Setup canvas canvas = document.getElementById('game-canvas'); ctx = canvas.getContext('2d'); resizeCanvas(); // Setup Socket.IO connection socket = io(); // Socket event handlers socket.on('init', ({ id, players: serverPlayers }) => { localPlayerId = id; players = serverPlayers; socket.emit('set-nickname', { nickname }); }); socket.on('new-player', ({ id, data }) => { players[id] = data; }); socket.on('update-player', ({ id, data }) => { if (players[id]) { players[id] = data; if (id === localPlayerId) { updateHealthDisplay(data.health); } } }); socket.on('remove-player', (id) => { delete players[id]; if (id === localPlayerId) { handleDeath(); } }); socket.on('player-fired', ({ shooterId }) => { if (shooterId !== localPlayerId) { // Visual feedback for shots (flash the shooter) const shooter = players[shooterId]; if (shooter) { // Flash effect logic here if desired } } }); // Setup event listeners window.addEventListener('resize', resizeCanvas); window.addEventListener('mousemove', handleMouseMove); window.addEventListener('mousedown', handleMouseDown); document.addEventListener('keydown', e => keys[e.key.toLowerCase()] = true); document.addEventListener('keyup', e => keys[e.key.toLowerCase()] = false); // Lock pointer on canvas click canvas.addEventListener('click', () => { canvas.requestPointerLock(); }); // Start game loop gameLoop(); } function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; } function handleMouseMove(e) { if (document.pointerLockElement === canvas) { mouseX += e.movementX; mouseY += e.movementY; } } function handleMouseDown(e) { if (e.button === 0 && document.pointerLockElement === canvas) { // Calculate direction based on mouse position const localPlayer = players[localPlayerId]; if (!localPlayer) return; // Find closest player in line of sight let closestPlayer = null; let closestDistance = Infinity; for (const [id, player] of Object.entries(players)) { if (id === localPlayerId) continue; const dx = player.x - localPlayer.x; const dy = player.y - localPlayer.y; const distance = Math.sqrt(dx * dx + dy * dy); // Check if player is in view angle (simplified) const angle = Math.atan2(dy, dx); const mouseAngle = Math.atan2(mouseY, mouseX); const angleDiff = Math.abs(angle - mouseAngle); if (distance < closestDistance && angleDiff < Math.PI / 4) { closestPlayer = id; closestDistance = distance; } } if (closestPlayer) { socket.emit('shoot', { hitPlayerId: closestPlayer }); } } } function handleDeath() { // Reset game document.exitPointerLock(); document.getElementById('game-container').style.display = 'none'; document.getElementById('nickname-container').style.display = 'block'; socket.disconnect(); } function updateHealthDisplay(newHealth) { health = newHealth; document.getElementById('health-value').textContent = health; } function gameLoop() { // Update local player position based on keyboard input if (localPlayerId && players[localPlayerId]) { const player = players[localPlayerId]; let dx = 0; let dy = 0; if (keys['w']) dy -= PLAYER_SPEED; if (keys['s']) dy += PLAYER_SPEED; if (keys['a']) dx -= PLAYER_SPEED; if (keys['d']) dx += PLAYER_SPEED; // Normalize diagonal movement if (dx !== 0 && dy !== 0) { dx *= 0.707; dy *= 0.707; } // Update position player.x += dx; player.y += dy; // Emit position update if moved if (dx !== 0 || dy !== 0) { socket.emit('move', { x: player.x, y: player.y }); } } // Clear canvas ctx.fillStyle = '#1a1a1a'; ctx.fillRect(0, 0, canvas.width, canvas.height); // Draw all players for (const [id, player] of Object.entries(players)) { // Draw player ctx.fillStyle = player.color; ctx.beginPath(); ctx.arc(player.x, player.y, PLAYER_SIZE / 2, 0, Math.PI * 2); ctx.fill(); // Draw nickname if (player.nickname) { ctx.fillStyle = 'white'; ctx.font = '14px Arial'; ctx.textAlign = 'center'; ctx.fillText(player.nickname, player.x, player.y - PLAYER_SIZE); } // Draw health bar const healthBarWidth = PLAYER_SIZE; const healthBarHeight = 4; const healthPercent = player.health / 100; ctx.fillStyle = 'red'; ctx.fillRect( player.x - healthBarWidth / 2, player.y + PLAYER_SIZE / 2 + 5, healthBarWidth, healthBarHeight ); ctx.fillStyle = 'green'; ctx.fillRect( player.x - healthBarWidth / 2, player.y + PLAYER_SIZE / 2 + 5, healthBarWidth * healthPercent, healthBarHeight ); } requestAnimationFrame(gameLoop); }
